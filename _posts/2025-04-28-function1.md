def sigma(limit): #작은 프로그램 단위 입출력 지정
    #limit: 매개변수, 매개체를 말한다. 함수 외부에서 함수 내부로 값을 전달하기 위한 목적이다 
    s = 0
    for i in range(1,limit + 1):
        s = s + i 
    return s 

print(sigma(10))
print(sigma(100))
print(sigma(1000))
